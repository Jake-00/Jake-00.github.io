```python
import ray
# import pandas as pd

# 初始化 Ray
ray.init()

# 创建数据集
ds = ray.data.from_items(
    [{"item": {"data": i}} for i in range(1000)]
)

# 第一阶段：标记奇偶性（使用正确的lambda格式）
ds_stage1 = ds.map_batches(
    lambda batch: {"item": [
        {"data": x["data"], "is_odd": x["data"] % 2 == 1} 
        for x in batch["item"]
    ]}
)

# 过滤出奇数（正确的filter写法）
odd_ds = ds_stage1.filter(
    lambda row: row["item"]["is_odd"]
)

# 过滤出偶数并平方
even_ds = ds_stage1.filter(
    lambda row: not row["item"]["is_odd"]
).map_batches(
    lambda batch: {"item": [
        {"data": x["data"]**2, "is_odd": x["is_odd"]} 
        for x in batch["item"]
    ]}
)

# 合并并计算最终结果
result = odd_ds.union(even_ds).map_batches(
    lambda batch: {"item": [
        {"data": x["data"]**2, "is_odd": x["is_odd"], "is_succ": True} 
        for x in batch["item"]
    ]}
)

# 收集并打印结果
print("处理结果（前20个）:")
count = 0
for batch in result.iter_batches(batch_size=100):
    for item in batch["item"]:
        print(f"数据: {item['data']:6d}, 是否奇数: {item['is_odd']}, 是否成功: {item['is_succ']}")
        count += 1
        if count >= 20:
            break
    if count >= 20:
        break

print(f"\n总共处理了 {result.count()} 条数据")

# 关闭 Ray
ray.shutdown()
```
